**Note:** `USTRUCT()` cannot have have `UFUNCTION()` methods
```cpp
USTRUCT(BlueprintType)
struct FCustomStruct{
  
  GENERATED_BODY()
  
  UPROPERTY(EditDefaultsOnly, BlueprintReadWrite)
  FName Name;

  UPROPERTY()
  int Value;

  // UFUNCTION() not useable in structs
  void Add() { Value++; }
};
```
