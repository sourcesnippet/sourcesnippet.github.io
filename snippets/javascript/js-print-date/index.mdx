```js
const date = new Date("2003-03-13");
const formatted_date = date.toLocaleDateString('en-GB', {
    weekday: 'long', // or 'short'
    day: 'numeric',
    month: 'long', // or 'short', 'numeric'
    year: 'numeric'
});
console.log(formatted_date); // Thursday 13 March 2003
```
