   public:
    UPROPERTY(EditAnywhere)

// MyPawn.h
UCLASS()
class TESTING_API AMyPawn : public APawn {
    GENERATED_BODY()

   public:
    AMyPawn();

    UPROPERTY(Instanced, EditAnywhere)
    UMyObject* MyObject;
};



// MyPawn.cpp
AMyPawn::AMyPawn(){
    MyObject = CreateDefaultSubobject<UMyObject>(TEXT("MyObject"));
}
