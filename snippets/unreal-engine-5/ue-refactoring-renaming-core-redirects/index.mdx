To be added in the file `/Config/DefaultEngine.ini`

```ini
[CoreRedirects]
+ClassRedirects=(OldName="OldPawn",NewName="NewPawn")

+StructRedirects=(OldName="OldStruct",NewName="NewStruct")

+EnumRedirects=(OldName="EOldEnum",NewName="ENewEnum",ValueChanges=(("ValueA","ValueB"),("Value1","Value2")))


+PropertyRedirects=(OldName="OldActor.OldIntProperty",NewName="NewActor.NewIntProperty")
+PropertyRedirects=(OldName="OldActor.OldFloatProperty",NewName="OldActor.NewFloatProperty")


+FunctionRedirects=(OldName="OldActor.OldFunction",NewName="NewActor.NewFunction")
+FunctionRedirects=(OldName="OldActor.OldFunction",NewName="OldActor.NewFunction")
```
